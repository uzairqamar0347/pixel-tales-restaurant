create table public.menu_items (
  id text not null,
  name text not null,
  description text not null,
  image_url text not null,
  category text not null,
  price numeric null,
  original_price numeric null,
  is_featured boolean null default false,
  price_small numeric null,
  price_medium numeric null,
  price_large numeric null,
  price_xl numeric null,
  constraint menu_items_pkey primary key (id)
) TABLESPACE pg_default;
create table public.categories (
  id bigint generated by default as identity not null,
  name text not null,
  display_order integer null,
  constraint categories_pkey primary key (id),
  constraint categories_name_key unique (name)
) TABLESPACE pg_default;
create table public.orders (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  items jsonb null,
  total_price numeric null,
  status text null default 'Pending'::text,
  customer_details jsonb null,
  constraint orders_pkey primary key (id)
) TABLESPACE pg_default;
-- ========= PART 2: SECURITY POLICIES (CORRECTED & SECURE) =========

-- First, we turn on the security system (RLS) for all tables.
-- This is a crucial step.
ALTER TABLE public.categories ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.menu_items ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.orders ENABLE ROW LEVEL SECURITY;


-- --- Policies for 'categories' table ---
-- This policy allows anyone to read the list of categories.
CREATE POLICY "Enable public read access for categories"
ON public.categories
FOR SELECT
USING (true);


-- --- Policies for 'menu_items' table ---
-- [SECURITY FIX] This policy allows anyone to READ the menu, but NOT change or delete it.
CREATE POLICY "Enable public read for menu items"
ON public.menu_items
FOR SELECT
USING (true);


-- --- Policies for 'orders' table ---
-- This policy allows the public (your website) to create new orders.
CREATE POLICY "Allow public insert for new orders"
ON public.orders
FOR INSERT
WITH CHECK (true);

-- This policy is for your future admin panel. It allows logged-in users to read orders.
CREATE POLICY "Allow authenticated users to read orders"
ON public.orders
FOR SELECT
USING (auth.role() = 'authenticated');
